architecture Behavioral of test_bench_SPI_Slave is
component SPI_Slave_Block is
	port(
		rst		:in std_logic;
		SCK 	:in std_logic;
		MOSI 	:in std_logic;
		MISO 	:out std_logic;
		SS 		:in std_logic;
		
		--Buffer'lar
		tx_data :in std_logic_vector(7 downto 0); --transmit buffer
		rx_data :out std_logic_vector(7 downto 0) --reciever buffer
	);
end component;

signal rst		: std_logic; 
signal SCK 	    : std_logic; 
signal MOSI 	: std_logic;
signal MISO 	: std_logic;
signal SS 		: std_logic;
signal tx_data  : std_logic_vector(7 downto 0);

signal cnt      : std_logic_vector(2 downto 0) := "000";
constant clock_periode : time:=20ns;
signal transfer_en : std_logic;

begin

Slave : SPI_Slave_Block port map(
    rst	 => rst	,
    SCK  => SCK ,
    MOSI => MOSI,
    MISO => MISO,
    SS 	 => SS,
    tx_data => tx_data
);

clock: process
begin
if(transfer_en = '1') then
    for k in 0 to 7 loop
        SCK <= '1';
        wait for clock_periode / 2;
        SCK <= '0';
        wait for clock_periode / 2;
    end loop;
else
    SCK <= '0';
end if;

end process;

--cnt_1 : process
--begin
--if(cnt = "111") then
--    cnt <= "000";
--else
--    if(transfer_en = '1' and rising_edge(SCK)) then
--        cnt <= cnt + "001";
--    end if;
--end if;
--end process;

process
begin

rst     <= '1';
MOSI    <= '0';
MISO    <= '0';
SS 	    <= '1';
tx_data <= "10011110";

wait for 30ns;

rst <= '0';

wait for 30ns;

SS <= '0';

wait for 30ns;

transfer_en <= '1';

if(rising_edge(SCK)) then
    if(cnt = "111") then
        cnt <= "000";
    end if;
MOSI <= tx_data(to_integer(unsigned(cnt)));
cnt <= cnt + "001";
end if;

transfer_en <= '0';

wait for 40ns;

wait;

end process;

end Behavioral;
